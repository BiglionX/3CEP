**示例代码**: 
```typescript
// 改进的错误处理模式
try {
  const response = await apiCall();
  if (!response.ok) {
    const error = await response.json();
    throw new Error(error.message);
  }
} catch (error: any) {
  // 分类错误处理
  if (error.message.includes('网络')) {
    setError('网络连接失败');
  } else if (error.message.includes('验证')) {
    setError('输入验证失败');
  } else {
    setError(`操作失败: ${error.message}`);
  }
}
```
